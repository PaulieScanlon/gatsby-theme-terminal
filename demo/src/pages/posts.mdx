import { mix } from '@theme-ui/color'

# posts

You can use the `<SourceList filter="posts" />` component from any `.mdx`. 

Here we use it to return all the `posts` by using the filter prop

<SourceList filter="posts">
  {posts => (
    <Flex
      sx={{
        flexWrap: "wrap",
        ml: theme => `-${theme.space[2]}px`,
        mr: theme => `-${theme.space[2]}px`,
      }}
    >
      {posts.map((edge, index) => {
        const {
          frontmatter: { title, featuredImage, tags },
          fields: { slug },
        } = edge.node
        return (
          <Box
            key={index}
            sx={{ mb: 3, width: ["100%", "100%", "50%", "33.333%"] }}
          >
            <Link href={slug} sx={{ textDecoration: "none", display: 'flex', height: '100%', ml: 2, mr: 2, }}>
              <Card
                sx={{
                  display: 'flex',
                  flexDirection: 'column',
  
                }}
              >
                {featuredImage && featuredImage.childImageSharp && (
                  <Image
                    src={featuredImage.childImageSharp.fluid.src}
                    alt={featuredImage.childImageSharp.fluid.originalName}
                  />
                )}
                <Box
                  sx={{
                    p: 3,
                    display: 'flex',
                    flexDirection: 'column',
                    flex: '1 1 auto',
                    justifyContent: 'space-between'
                  }}
                >
                <Box
                  sx={{mb: 4}}
                >
                  <Heading variant="styles.h4" sx={{ color: "text" }}>
                    {title}
                  </Heading>
                  {tags &&
                    tags.map((tag, index) => (
                      <Badge
                        key={index}
                        sx={{
                          mr: 2,
                          mb: 2,
                          color: mix(
                            "muted",
                            "primary",
                            `${index / tags.length}`
                          ),
                          borderColor: mix(
                            "muted",
                            "primary",
                            `${index / tags.length}`
                          ),
                        }}
                        variant="primary"
                      >
                        {tag}
                      </Badge>
                    ))}
                  </Box>
                  <Box>
                  <Text sx={{ textDecoration: "underline" }}>View post</Text>
                  </Box>
                </Box>
              </Card>
            </Link>
          </Box>
        )
      })}
    </Flex>
  )}
</SourceList>
