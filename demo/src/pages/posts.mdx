# posts

You can use the `<SourceList filter="posts" />` component from any `.mdx`. 

Here we use it to return all the `posts` by using the filter prop

<SourceList filter="posts">
  {posts => (
    <Flex
      sx={{
        flexWrap: "wrap",
        ml: theme => `-${theme.space[2]}px`,
        mr: theme => `-${theme.space[2]}px`,
      }}
    >
      {posts.map((edge, index) => {
        const {
          frontmatter: { title, featuredImage, tags, date },
          excerpt,
          fields: { slug },
        } = edge.node
        return (
          <Box
            key={index}
            sx={{ display: 'flex', flex: '1 1 auto', flexDirection: 'column', mb: 3, width: ["100%", "100%", "50%", "33.333%"], outline: '1px solid red' }}
          >
             <Link href={slug} sx={{ textDecoration: "none", display: 'flex', flex: '1 1 auto', flexDirection: 'column' }}>
    
               <Card sx={{ display: 'flex', flex: '1 1 auto', flexDirection: 'column', outline: '1px solid blue' }}>
                {/* {featuredImage && featuredImage.childImageSharp && (
                  <Box sx={{ position: 'relative', overflow: 'hidden' }}>
                    <Image
                      src={featuredImage.childImageSharp.fluid.src}
                      alt={featuredImage.childImageSharp.fluid.originalName}
                    />
                  </Box>
                )} */}
                  <Box sx={{ 
                    display: 'flex', 
                    flex: '1 1 auto', 
                    flexDirection: 'column', 
                    border: '1px solid yellow', 
                    p: 3}}>
                  <Heading variant="styles.h4" sx={{ color: "text" }}>
                    {title}
                  </Heading>
                   <Text sx={{mb: 1, color: 'muted'}}>{date}</Text>
                   <Text sx={{mb: 1, border: '1px solid green', color: 'text'}}>{excerpt}</Text>
                  </Box>
                  <Box sx={{p: theme => `0px ${theme.space[3]}px`}}>
                  <Text>View Post</Text>
                  </Box>
              </Card>
   
            </Link> 
          </Box>
        )
      })}
    </Flex>
  )}
</SourceList>
