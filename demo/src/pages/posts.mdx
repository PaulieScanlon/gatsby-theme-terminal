import { mix } from '@theme-ui/color'

# posts

You can use the `<SourceList filter="posts" />` component from any `.mdx`. 

Here we use it to return all the `posts` by using the filter prop

<SourceList filter="posts">
  {posts => (
    <Flex
      sx={{
        flexWrap: "wrap",
        ml: theme => `-${theme.space[2]}px`,
        mr: theme => `-${theme.space[2]}px`,
      }}
    >
      {posts.map((edge, index) => {
        const {
          frontmatter: { title, featuredImage, tags, date },
          excerpt,
          fields: { slug },
        } = edge.node
        return (
          <Box
            key={index}
            sx={{ display: 'flex', mb: 3, width: ["100%", "100%", "50%", "33.333%"], pl: 2, pr: 2 }}
          >
            <Link href={slug} sx={{ textDecoration: "none", display: 'flex', flex: '1 1 auto' }}>
             <Card sx={{display: 'flex', flexDirection: 'column'}}>
                {featuredImage && featuredImage.childImageSharp && (
                  <Box sx={{ position: 'relative', overflow: 'hidden' }}>
                    <Image
                      src={featuredImage.childImageSharp.fluid.src}
                      alt={featuredImage.childImageSharp.fluid.originalName}
                    />
                  </Box>
                )}
                <Box
                  sx={{
                    flex: '1 1 auto',
                    p: 3,
                  }}
                >
                  <Heading variant="styles.h4" sx={{ color: "text" }}>
                    {title}
                  </Heading>
                  <Text sx={{mb: 2, color: 'muted'}}>{date}</Text>
                  <Text sx={{mb: 2, color: 'text'}}>{excerpt}</Text>
                  <Box sx={{display: 'block' }}>
                  {tags &&
                    tags.map((tag, index) => (
                      <Badge
                        key={index}
                        sx={{
                          mr: 2,
                          mb: 2,
                          color: mix(
                            "muted",
                            "primary",
                            `${index / tags.length}`
                          ),
                          borderColor: mix(
                            "muted",
                            "primary",
                            `${index / tags.length}`
                          ),
                        }}
                        variant="primary"
                      >
                        {tag}
                      </Badge>
                    ))}
                  </Box>
                </Box>
              </Card>
            </Link>
          </Box>
        )
      })}
    </Flex>
  )}
</SourceList>
